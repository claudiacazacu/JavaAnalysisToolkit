// Even / odd demo with modulo and &&
int x = 5;
int evenCount = 0;
int oddCount = 0;

while (x > 0) {
    if (x % 2 == 0 && x > 1) {
        evenCount = evenCount + 1;
        print "even";
    } else {
        oddCount = oddCount + 1;
        print "odd";
    }

    x = x - 1;
}

print evenCount;
print oddCount;
